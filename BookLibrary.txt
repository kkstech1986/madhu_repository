    def borrowBook(self,bookName):
        if  bookName in self.books:
            
            print(f"you have been issued {bookName}.please keep it safe and return if within 30 days")
                 
            self.books.remove(bookName)
            self.rBook.append(bookName)
            return True                   
        else:        
            print("sorry, this book has already been issue sumbody please waite unlite the book are returen ")
            return False
                        
    def returnBook(self,bookName):
        if bookName in self.rBook :
            self.books.append(bookName)
            self.rBook.remove(bookName)         
        else:
            print(f"this book are not match our book return are {self.rBook} ")
